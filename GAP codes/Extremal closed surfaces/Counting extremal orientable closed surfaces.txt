gap> #This code describes all the possible Dirichlet domains of extremal orientable closed surfaces of genus g, centered at a point z that projects into an extremal disc center of the corresponding extremal surface.
gap> g:=2;; #Input: can be changed to our desired value of genus.
gap> m:=12*g-6;;
gap> #
gap> F:= FreeGroup("x","y");;
gap> x:=F.1;; y:=F.2;;
gap> G:=F/[(x)^2, (y)^3, (x*y)^m];; #Fuchsian triangle group of signature (2,3,m).
gap> torsion:=[x,y,y^2];; #List of torsion elements in G (up to conjugation).
gap> for j in [1..(m-1)] do;
> Append(torsion, [(x*y)^j]);
> od;
gap> #
gap> #We want to find the index m subgroups of G that do not contain any element in the list "torsion", so they are torsion-free.
gap> index:=m;;
gap> listofsubgroups_0:= LowIndexSubgroupsFpGroup(G,TrivialSubgroup(G),index,torsion);;
gap> listofsubgroups:=[];;
gap> for k in [1..Size(listofsubgroups_0)] do;
> if Index(G, listofsubgroups_0[k])=index and #Subgroups with the desired index m.
> Size(MovedPoints(List(CosetTable(G, listofsubgroups_0[k]), PermList)[1]))=index and #The fundamental domain is a m-gon whose sides are identified by pairs.
> CycleStructurePerm(List(CosetTable(G, listofsubgroups_0[k]), PermList)[3])[2]=m/2+1-2*g then #The number of vertices of the fundamental domain is the desired one.
> Append(listofsubgroups,[listofsubgroups_0[k]]);
> fi;
> od;
gap> #
gap> #Calculation of the associated monodromy morphisms:
gap> listofmonodromies_0:=[];;
gap> for j in [1..Size(listofsubgroups)] do;
> Append(listofmonodromies_0, List(CosetTable(G,listofsubgroups[j]){[1,3]}, PermList)); #Permutations given by the image of the generators x and y by the monodromy morphisms.
> od;
gap> listofmonodromies:=[];;
gap> for i in [1..Size(listofmonodromies_0)/2] do;
> Append(listofmonodromies, [[listofmonodromies_0[2*i-1], listofmonodromies_0[2*i], ((listofmonodromies_0[2*i-1])*(listofmonodromies_0[2*i]))^(-1)]]);
> od;
gap> #For clearness, we are going to normalize the third permutation via conjugation by a fixed permutation, which is equivalent to re-label the sides of the fundamental domain.
gap> normalized_list:=[];;
gap> for j in [1..m-1] do;
> Add(normalized_list,j+1);
> od;
gap> Add(normalized_list,1);
gap> normalized:=PermList(normalized_list);;
gap> listofnormalizedmonodromies:=[];;
gap> for k in [1..Size(listofmonodromies)] do;
> norm:=RepresentativeAction(SymmetricGroup(index),normalized,listofmonodromies[k][3]);
> Append(listofnormalizedmonodromies, [[norm*listofmonodromies[k][1]*norm^(-1),norm*listofmonodromies[k][2]*norm^(-1),norm*listofmonodromies[k][3]*norm^(-1)]]);
> #Now, the third entry of each monodromy 3-tuple is the permutation called "normalized", so the sides of the fundamental domain are labelled counterclockwise.
> od;
gap> side_pairings:=[];;
gap> for k in [1..Size(listofnormalizedmonodromies)] do;
> Append(side_pairings, [[listofnormalizedmonodromies[k][1]]]);
> od;
gap> Print("There exists exactly ", Size(listofnormalizedmonodromies), " extremal disc configurations where the associated surface is extremal, orientable, closed and \nhas genus ", g,".");
gap> Print("The corresponding fundamental domains are given by some regular ", m, "-agons with all their inner angles of measure 2pi/3 \nand whose sides are labelled counterclockwise.");
gap> Print("Their side-pairing patterns are given by the following permutations: ", "\n", side_pairings);
